								{\infer[\land I]{[(0 + 0 = 0) \land \forall x \ ((0 + x = x) \to (0 + s(x) = s(x)))]}
									{\infer[\forall E]{(0 + 0 = 0)}

									&
									\infer[\forall I]{\forall x \ [(0 + x = x) \to (0 + s(x) = s(x))]}

											{\infer[=E]{(0 + s(n) = s(n))}

							{\infer[\forall I]{\forall y \ [y+s(n) = s(n)+y]}
								{\infer[=E]{a + s(n) = s(n) + a}
									{\infer[=E]{a + s(n) = s(n + a)}
										{\infer[\forall E]{a + s(n) = s(a+n)}
											{\forall x \ \forall y \ (x + s(y) = s(x + y))}
										&
										\infer[\forall E]{a + n = n + a}
											{\infer[\IH]{\forall y \ [y + n = n + y]}{}}}
									&
									\infer[\forall E]{s(n) + a = s(n+a)}
										{\infer[\IND]{\forall x \ \forall y \ [s(y) + x = s(y+x)]}
										{\infer[\land I]{[\forall y \ [s(y) + 0 = s(y+0)] \land \forall x \ (\forall y \ [s(y) + x = s(y+x)] \to \forall y \ [s(y) + s(x) = s(y+s(x))])]}
											{\infer[\forall I]{\forall y \ [s(y) + 0 = s(y+0)]}
												{\infer[=E]{s(a) + 0 = s(a + 0)}
													{\infer[=E]{s(a) + 0 = s(a)}
														{\infer[=I]{s(a)=s(a)}{}
														&
														\infer[\forall E]{s(a)+0=s(a)}{\forall x \ (x + 0 = x)}}
													&
													\infer[\forall E]{a = a + 0}{\forall x \ (x + 0 = x)}}}
											&
											\infer[\forall I]{\forall x \ [(\forall y \ [s(y) + n = s(y+n)] \to \forall y \ [s(y) + s(n) = s(y+s(n))])]}
												{\infer[\to I, \IH]{(\forall y \ [s(y) + n = s(y+n)] \to \forall y \ [s(y) + s(n) = s(y+s(n))])}
													{\infer[\forall I]{\forall y \ s(y)+s(n) = s(y + s(n))}
														{\infer[=E]{s(a)+s(n) = s(a + s(n))}
															{\infer[]{s(a)+s(n)=s(s(a+n))}
														{\infer[\forall E]{s(a)+s(n)=s(s(a)+n)}{\forall x \ \forall y \ (x + s(y) = s(x + y))}
														&
														\infer[\forall E]{s(a) + n = s(a+n)}
															{\infer[\IH]{\forall y \ (s(y) + n = s(a+n))}{}}}
													&
													\infer[\forall E]{a + s(n) = s(a + n)}{\forall x \ \forall y \ (x + s(y) = s(x + y))}}}}}}}}}}}}}}
